# --- DietPi-Automation for Takeout Processor Appliance v1.1 ---

# 1. System Settings
AUTO_SETUP_NET_HOSTNAME=takeout-processor
AUTO_SETUP_TIMEZONE=America/New_York # User should change this
AUTO_SETUP_KEYBOARD_LAYOUT=us

# 2. Automatic Software Installation
# Install only the bare minimum required by the python script itself.
# The setup script will be downloaded, not cloned.

# 3. The Custom User Script
# This is the core of the automation. DietPi will download and execute this
# exact script after the initial software installation.

# 4. Final Automation Settings
AUTO_SETUP_AUTOSTART_TARGET_INDEX=0 # Autologin as root
AUTO_SETUP_AUTOMATED=1
AUTO_SETUP_ACCEPT_LICENSE=1
