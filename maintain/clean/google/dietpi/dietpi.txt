# --- DietPi-Automation Settings for Takeout Processor Appliance ---

# 1. System Settings
AUTO_SETUP_NET_HOSTNAME=takeout-processor
AUTO_SETUP_TIMEZONE=America/New_York # User should change this
AUTO_SETUP_KEYBOARD_LAYOUT=us

# 2. Networking (Assuming Ethernet DHCP)
AUTO_SETUP_NET_USESTATIC=0

# 3. Automatic Software Installation
# We need Git (ID 88), Python 3 (ID 138), and jdupes (custom APT install)
AUTO_SETUP_INSTALL_SOFTWARE_ID=88
AUTO_SETUP_INSTALL_SOFTWARE_ID=138
# Install jdupes via APT
AUTO_SETUP_APT_INSTALLS=jdupes

# 4. The "Secret Sauce": The User Script
# This tells DietPi to download and run our custom setup script AFTER software is installed.
# This script will set up our application and the autostart loop.
AUTO_SETUP_CUSTOM_SCRIPT_EXEC=https://raw.githubusercontent.com/4ndr0666/scr/main/path/to/our/setup_script.sh #<-- Needs a real URL

# 5. Final Automation Settings
# Set the user to auto-login (e.g., root is 0, dietpi is 2)
AUTO_SETUP_AUTOSTART_TARGET_INDEX=0
# Tell the system the setup is automated and to accept licenses
AUTO_SETUP_AUTOMATED=1
AUTO_SETUP_ACCEPT_LICENSE=1
